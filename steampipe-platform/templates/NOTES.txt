Prerequisites:
{{- if .Values.providers.aws.enabled }}
  AWS - Ensure the credentials secret exists before installing:

    kubectl -n {{ .Values.namespace }} create secret generic {{ .Values.providers.aws.credentialsSecretName }} \
      --from-literal=AWS_ACCESS_KEY_ID=<your-key-id> \
      --from-literal=AWS_SECRET_ACCESS_KEY=<your-secret-key> \
      --from-literal=AWS_REGION=<your-region>
{{- end }}
{{- if .Values.providers.gcp.enabled }}
  GCP - Ensure the credentials secret exists before installing:

    kubectl -n {{ .Values.namespace }} create secret generic {{ .Values.providers.gcp.credentialsSecret }} \
      --from-file=credentials.json=<path-to-service-account-key.json>
{{- end }}
{{- if .Values.providers.kubernetes.enabled }}
{{- if .Values.providers.kubernetes.clusters }}
  Kubernetes - Create a kubeconfig Secret for each target cluster before installing:
{{- range .Values.providers.kubernetes.clusters }}

    kubectl -n {{ $.Values.namespace }} create secret generic {{ .kubeconfigSecret }} \
      --from-file={{ .kubeconfigKey | default "kubeconfig" }}=<path-to-kubeconfig>
{{- end }}

  Each kubeconfig should contain credentials with read access to the target cluster.
{{- end }}
{{- end }}

Powerpipe:
  Access the dashboard:

    kubectl -n {{ .Values.namespace }} port-forward svc/{{ include "steampipe-platform.fullname" . }}-powerpipe {{ .Values.powerpipe.port }}:{{ .Values.powerpipe.port }}

  Then open: http://localhost:{{ .Values.powerpipe.port }}

Check refresh status:

    kubectl -n {{ .Values.namespace }} exec -it {{ include "steampipe-platform.dbClusterName" . }}-1 -- psql -U postgres -d steampipe -c "
      SELECT * FROM cache.refresh_log ORDER BY last_refreshed DESC;
    "

Cached Tables (per account):
{{- range .Values.providers.aws.cachedTables }}
  - {{ .name }} (schedule: {{ .schedule }})
{{- end }}

On-Demand Refresh:

    kubectl -n {{ .Values.namespace }} exec -it {{ include "steampipe-platform.dbClusterName" . }}-1 -- psql -U postgres -d steampipe -c "
      SELECT cache.refresh_and_log('<account_name>_<table_name>');
    "

Manual table removal (multi-account):

    kubectl -n {{ .Values.namespace }} exec -it {{ include "steampipe-platform.dbClusterName" . }}-1 -- psql -U postgres -d steampipe -c "
      SELECT cron.unschedule('refresh-<account_name>-<table_name>');
      DROP MATERIALIZED VIEW IF EXISTS cache.<account_name>_<table_name> CASCADE;
    "

  After removing, recreate the aggregation view or run helm upgrade to rebuild it.
